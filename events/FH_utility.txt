country_event = { #GRI annexes BOI
	id = griffonia_activity.1
	immediate = {log = "[GetDateText]: [Root.GetName] event griffonia_activity.1"}
	title = griffonia_activity.1.t
	desc = griffonia_activity.1.d
	hidden = yes
	
	trigger = {
		tag = GRI
		date > 1051.1.1
	}
	
	fire_only_once = yes
	
	option = {
		name = griffonia_activity.1.a
		annex_country = {
			target = BOI
			transfer_troops = yes
		}
		every_state = {
			limit = {
				is_core_of = BOI
			}
			add_core_of = ROOT
		}
	}
}
