				limit = { infrastructure > 1 }

					type = infrastructure
					damage = 2
				}
			}
			ELSE_IF = {
				limit = { 
					AND = {
						infrastructure > 0
						infrastructure < 2
					}
				}
				damage_building = {
					type = infrastructure
					damage = 1
				}
			}
			IF = {
				limit = { arms_factory > 0 }
				damage_building = {
					type = arms_factory
					damage = 1
				}
			}
			ELSE_IF = {
				limit = { industrial_complex > 0 }
				damage_building = {
					type = industrial_complex
					damage = 1
				}
			}
		}
		hidden_effect = {
			IF = {
				limit = { check_variable = { terrorism_counter = 0 } }
				set_variable = { terrorism_counter = 0 }
			}
			IF = {
				limit = { check_variable = { terrorism_counter > 1 } }

					id = southeast.500
