on_actions = {
    on_startup = {
        effect = {
            every_country = {
                limit = {
                    is_griffon_continent_country = yes
                    has_custom_focus_tree_content = no
                }
                load_focus_tree = generic_griffon_focus
            }
            every_country = {
                limit = {
                    is_zebrican_continent_country = yes
                    has_custom_focus_tree_content = no
                }
                load_focus_tree = generic_zebra_focus
            }
            every_country = {
                limit = {
                    is_griffon_continent_country = no
                    is_zebrican_continent_country = no
                    has_custom_focus_tree_content = no
                }
                load_focus_tree = generic_focus
            }
        }
    }
}
