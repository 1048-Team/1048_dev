faction_goal_control_griffonia = {
    name = control_griffonia_name
    description = control_griffonia_desc
    category = long_term
    group = FOCUS_FILTER_ANNEXATION

    visible = {
        FROM = {
            OR = {
                has_cosmetic_tag = EQC_maresoc
                is_in_faction_with = EQC
            }
        }
    }
    completed = {
        all_collection_elements = {
            collection = {
                input = collection:all_griffonia_states
            }
            is_controlled_by_ROOT_or_ally = yes
        }
    }
    complete_effect = {
        add_faction_power_projection = 500
        add_faction_initiative = 1
		custom_effect_tooltip = generic_skip_one_line_tt
		custom_effect_tooltip = ALL_FACTION_MEMBERS
		every_faction_member = {
			add_political_power = 100
		}
    }
}
