SOL_ordo_special_effect = {
	IF = {
		limit = {
			check_ordo_selection_special = yes
		}
		IF = {
			limit = {
				check_variable = { ordo_selection = 1 }
			}
			add_to_variable = { EQS.ordo_loyalty_array^var:ordo_selection = 7 }
			SOL_ordo_set_flag_special = yes
		}
		ELSE_IF = {
			limit = {
				check_variable = { ordo_selection = 2 }
			}
			add_to_variable = { EQS.ordo_loyalty_array^var:ordo_selection = 10 }
			clamp_variable = { 
				var = EQS.ordo_loyalty_array^var:ordo_selection
				min = 0
				max = 100
			}
			SOL_ordo_set_flag_special = yes
		}
	}
}
