SOL_ordo_special_effect = {
	IF = {
		limit = {
			AND = {
				NOT = {
					has_idea = EQS_great_ordo_work_funding
				}
				check_variable = {
					var = SOL_imperial_authority_pool
					value = 10
					compare = greater_than_or_equals
				}
			}
		}
		IF = {
			limit = {
				AND = {
					OR = {
						AND = {
							check_variable = { ordo_selection = 22 }
							SOL_owns_northern_tenuis_states = yes
						}
						NOT = { check_variable = { ordo_selection = 22 } }
					}
				}
			}
			SOL_special_production_effect = yes
			add_to_variable = { global.ordo_loyalty_array^var:ordo_selection = 25 }
			clamp_variable = { 
				var = global.ordo_loyalty_array^var:ordo_selection
				min = 0
				max = 100
			}
			add_timed_idea = {
				idea = EQS_great_ordo_work_funding
				days = 180
			}
			add_to_variable = { SOL_imperial_authority_pool = -10 }
			SOL_ordo_set_flag_special = yes
			IF = {
				limit = {
					check_variable = {
						var = SOL_imperial_authority_pool
						value = 75
						check = greater_than_or_equals
					}
				}
				SOL_high_authority_derived_stats = yes
			}
			IF = {
				limit = {
					check_variable = {
						var = SOL_imperial_authority_pool
						value = 25
						check = less_than_or_equals
					}
				}
				SOL_low_authority_derived_stats = yes
			}
		}
		ELSE = {
			country_event = solarordo.26
		}
	}
}

SOL_ordo_loyalty_bonus = {
	IF = {
		limit = {
			ROOT = {
				is_ai = no
			}
		}
		IF = {
			limit = {
				check_variable = { SOL_ordo_average_loyalty > 84 }
			}
			SOL_ordo_loyalty_very_high = yes
		}
		ELSE_IF = {
			limit = {
				AND = {
					check_variable = { SOL_ordo_average_loyalty < 85 }
					check_variable = { SOL_ordo_average_loyalty > 60 }
				}
			}
			SOL_ordo_loyalty_high = yes
		}
		ELSE_IF = {
			limit = {
				AND = {
					check_variable = { SOL_ordo_average_loyalty < 61 }
					check_variable = { SOL_ordo_average_loyalty > 44 }
				}
			}
			SOL_ordo_loyalty_average = yes
		}
		ELSE_IF = {
			limit = {
				AND = {
					check_variable = { SOL_ordo_average_loyalty < 45 }
					check_variable = { SOL_ordo_average_loyalty > 20 }
				}
			}
			SOL_ordo_loyalty_low = yes
		}
		ELSE_IF = {
			limit = {
				check_variable = { SOL_ordo_average_loyalty < 21 }
			}
			SOL_ordo_loyalty_very_low = yes
		}
	}
	ELSE = {
		SOL_ordo_loyalty_average = yes
	}
}
